import cv2 import numpy as np from pyzbar.pyzbar import decode from gtts import gTTS import pygame import os import time # --- 初始化語音引擎 --- pygame.mixer.init() def speak(text): """將文字轉為語音並播放""" filename = "voice.mp3" try: if os.path.exists(filename): os.remove(filename) tts = gTTS(text=text, lang='zh-tw') tts.save(filename) pygame.mixer.music.load(filename) pygame.mixer.music.play() except Exception as e: print(f"語音播放失敗: {e}") # --- 遊戲設定 --- levels = [ {"name": "青蛙", "color_name": "綠色", "qr_target": "Green", "bgr": (0, 255, 0), "shape": "circle"}, {"name": "蘋果", "color_name": "紅色", "qr_target": "Red", "bgr": (0, 0, 255), "shape": "rect"}, {"name": "天空", "color_name": "藍色", "qr_target": "Blue", "bgr": (255, 0, 0), "shape": "rect"} ] current_level_idx = 0 is_colored = False last_speech_time = 0 def get_game_image(level, colored=False): """動態生成遊戲圖片 (黑白或彩色)""" img = np.ones((400, 400, 3), dtype=np.uint8) * 240 # 淺灰色背景 color = level["bgr"] if colored else (100, 100, 100) # 彩色或深灰色 if level["shape"] == "circle": cv2.circle(img, (200, 200), 120, color, -1) cv2.circle(img, (200, 200), 120, (0, 0, 0), 3) # 輪廓 else: cv2.rectangle(img, (100, 100), (300, 300), color, -1) cv2.rectangle(img, (100, 100), (300, 300), (0, 0, 0), 3) return img # --- 主程式 --- cap = cv2.VideoCapture(0) print("遊戲開始！按 'q' 退出，按 'n' 跳到下一關。") while True: ret, frame = cap.read() if not ret: break frame = cv2.flip(frame, 1) # 鏡像畫面 level = levels[current_level_idx] # 每 5 秒自動提醒一次語音 (避免重複播放過於頻繁) if time.time() - last_speech_time > 5 and not is_colored: speak(f"請拿出{level['color_name']}的 QR Code，幫{level['name']}塗上顏色") last_speech_time = time.time() # 1. 偵測 QR Code qr_codes = decode(frame) for qr in qr_codes: data = qr.data.decode('utf-8').strip() (x, y, w, h) = qr.rect cv2.rectangle(frame, (x, y), (x + w, y + h), (0, 255, 255), 2) # 判定答案 if data.lower() == level["qr_target"].lower(): if not is_colored: is_colored = True speak(f"答對了！{level['name']}變成{level['color_name']}了！") else: cv2.putText(frame, "Try Again!", (x, y-10), cv2.FONT_HERSHEY_SIMPLEX, 0.7, (0, 0, 255), 2) # 2. 準備遊戲畫面 (右側 UI) game_img = get_game_image(level, colored=is_colored) # 3. 結合攝像頭與遊戲畫面 # 將遊戲圖案縮放並放在畫面中央或角落 h, w, _ = frame.shape gh, gw, _ = game_img.shape frame[10:10+gh, 10:10+gw] = game_img # 顯示文字指示 status_text = f"Target: {level['color_name']} {level['qr_target']}" cv2.putText(frame, status_text, (20, 450), cv2.FONT_HERSHEY_COMPLEX, 1, (255, 255, 255), 4) cv2.putText(frame, status_text, (20, 450), cv2.FONT_HERSHEY_COMPLEX, 1, (0, 0, 0), 2) cv2.imshow("Color QR Game", frame) # 按鍵控制 key = cv2.waitKey(1) & 0xFF if key == ord('q'): break elif key == ord('n') or (is_colored and key == 32): # 按 N 或 答對後按空白鍵下一關 current_level_idx = (current_level_idx + 1) % len(levels) is_colored = False last_speech_time = 0 cap.release() cv2.destroyAllWindows()
